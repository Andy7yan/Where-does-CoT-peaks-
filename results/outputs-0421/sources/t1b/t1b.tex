\newcommand{\PaintCell}[3]{%
  \pgfmathsetmacro{\xa}{#1 - 0.5}%
  \pgfmathsetmacro{\xb}{#1 + 0.5}%
  \pgfmathsetmacro{\ya}{#2 - 0.5}%
  \pgfmathsetmacro{\yb}{#2 + 0.5}%
  \path[fill=#3, draw=white, line width=0.28pt]
    (axis cs:\xa,\ya) rectangle (axis cs:\xb,\yb);%
}

\newcommand{\DrawBaseGrid}{%
  \pgfmathtruncatemacro{\LMinInt}{\LMin}%
  \pgfmathtruncatemacro{\LMaxInt}{\LMax}%
  \pgfmathtruncatemacro{\StepMaxInt}{\StepMax}%
  \foreach \yy in {\LMinInt,...,\LMaxInt}{%
    \foreach \xx in {1,...,\StepMaxInt}{%
      \ifnum\xx>\yy
        \PaintCell{\xx}{\yy}{InvalidCell}%
      \else
        \PaintCell{\xx}{\yy}{EmptyCell}%
      \fi
    }%
  }%
}

% Map a normalised scalar in [-1, 1] to the requested diverging palette.
\newcommand{\NormToFill}[1]{%
  \pgfmathsetmacro{\NormVal}{max(-1,min(1,#1))}%
  \pgfmathsetmacro{\MixPct}{100*abs(\NormVal)}%
  \ifdim \NormVal pt < 0pt
    \edef\ThisFill{ZeroColor!\MixPct!NegOneColor}%
  \else
    \edef\ThisFill{ZeroColor!\MixPct!PosOneColor}%
  \fi
}

\newcommand{\DrawNLDDCells}[1]{%
  \pgfplotstablegetrowsof{#1}%
  \pgfmathtruncatemacro{\RowMax}{\pgfplotsretval - 1}%
  \pgfplotsinvokeforeach{0,...,\RowMax}{%
    \pgfplotstablegetelem{##1}{L}\of{#1}\edef\CellLRaw{\pgfplotsretval}%
    \pgfplotstablegetelem{##1}{step}\of{#1}\edef\CellStepRaw{\pgfplotsretval}%
    \pgfmathtruncatemacro{\CellL}{\CellLRaw}%
    \pgfmathtruncatemacro{\CellStep}{\CellStepRaw}%
    % NLDD at step=1 is intentionally blank by spec.
    \ifnum\CellStep>1
      \pgfplotstablegetelem{##1}{mean_nldd}\of{#1}\edef\CellValRaw{\pgfplotsretval}%
      \NormToFill{\CellValRaw / \NLDDClip}%
      \PaintCell{\CellStep}{\CellL}{\ThisFill}%
    \fi
  }%
}

\newcommand{\DrawTASCells}[1]{%
  \pgfplotstablegetrowsof{#1}%
  \pgfmathtruncatemacro{\RowMax}{\pgfplotsretval - 1}%
  \pgfplotsinvokeforeach{0,...,\RowMax}{%
    \pgfplotstablegetelem{##1}{L}\of{#1}\edef\CellLRaw{\pgfplotsretval}%
    \pgfplotstablegetelem{##1}{step}\of{#1}\edef\CellStepRaw{\pgfplotsretval}%
    \pgfmathtruncatemacro{\CellL}{\CellLRaw}%
    \pgfmathtruncatemacro{\CellStep}{\CellStepRaw}%
    \pgfplotstablegetelem{##1}{mean_tas}\of{#1}\edef\CellValRaw{\pgfplotsretval}%
    % mean_tas is assumed in [0,1]; remap to [-1,1] so the same diverging palette is used.
    \NormToFill{2*\CellValRaw - 1}%
    \PaintCell{\CellStep}{\CellL}{\ThisFill}%
  }%
}

\newcommand{\DrawManualColourbar}[4]{%
  % #1 x, #2 y, #3 width, #4 label
  \begin{scope}[shift={(#1,#2)}]
    \shade[left color=NegOneColor, middle color=ZeroColor, right color=PosOneColor]
      (0,0) rectangle (#3,0.22);
    \draw[AxisGrey, line width=0.3pt] (0,0) rectangle (#3,0.22);
    \node[font=\scriptsize, anchor=north] at (0,0) {-1};
    \node[font=\scriptsize, anchor=north] at ({0.5*#3},0) {0};
    \node[font=\scriptsize, anchor=north] at (#3,0) {1};
    \node[font=\scriptsize, anchor=south] at ({0.5*#3},0.26) {#4};
  \end{scope}
}
