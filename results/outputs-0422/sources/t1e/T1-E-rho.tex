\newcommand{\ResolveCSVPath}[3]{%
  \ifx#2\@empty
    \IfFileExists{#3/#1}{\edef#2{#3/#1}}{}%
  \fi
}
\newcommand{\ResolveCSV}[2]{%
  \def#2{}%
  \ResolveCSVPath{#1}{#2}{.}%
  \ResolveCSVPath{#1}{#2}{data}%
  \ResolveCSVPath{#1}{#2}{csv}%
  \ResolveCSVPath{#1}{#2}{plots}%
  \ResolveCSVPath{#1}{#2}{plot}%
  \ResolveCSVPath{#1}{#2}{fig_data}%
  \ResolveCSVPath{#1}{#2}{fig-data}%
  \ResolveCSVPath{#1}{#2}{tables}%
  \ResolveCSVPath{#1}{#2}{stats}%
  \ResolveCSVPath{#1}{#2}{analysis}%

  \ResolveCSVPath{#1}{#2}{t1b_splits}%
  \ResolveCSVPath{#1}{#2}{t1c_splits}%

  \ifx#2\@empty
    \PackageError{t1e-rho}{Could not locate `#1'}%
      {Put the file in the project root or add its folder to \string\ResolveCSV.}%
  \fi
}
\makeatother

\newcommand{\FmtTwo}[1]{\pgfmathprintnumber[fixed,precision=2]{#1}}

\begin{document}

\ResolveCSV{plot_src_fig_a_rho.csv}{\FigARawPath}
\ResolveCSV{plot_src_fig_a_rho_stats.csv}{\FigAStatsPath}
\pgfplotstableread[col sep=comma]{\FigARawPath}\LoadedTable
\pgfplotstableread[col sep=comma]{\FigAStatsPath}\FigAStats
\pgfplotstablegetelem{0}{median_rho}\of{\FigAStats}\edef\MedianRho{\pgfplotsretval}
\pgfplotstablegetelem{1}{n}\of{\FigAStats}\edef\NMediumA{\pgfplotsretval}
\pgfplotstablegetelem{2}{n}\of{\FigAStats}\edef\NHardA{\pgfplotsretval}
\pgfmathsetmacro{\MedianRhoFrac}{(\MedianRho + 1.0)/2.0}
\gdef\HardRhoData{}
\gdef\MediumRhoData{}
\pgfplotstableforeachcolumnelement{rho}\of\LoadedTable\as\ThisRho{%
  \pgfplotstablegetelem{\pgfplotstablerow}{difficulty}\of{\LoadedTable}%
  \edef\ThisDiff{\pgfplotsretval}%
  \ifnum\pdfstrcmp{\ThisDiff}{hard}=0\relax
    \xdef\HardRhoData{\HardRhoData (\ThisRho,0)}%
  \fi
  \ifnum\pdfstrcmp{\ThisDiff}{medium}=0\relax
    \xdef\MediumRhoData{\MediumRhoData (\ThisRho,0)}%
  \fi
}
