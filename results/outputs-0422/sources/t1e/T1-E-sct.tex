\makeatletter
\newcommand{\ResolveCSVPath}[3]{%
  \ifx#2\@empty
    \IfFileExists{#3/#1}{\edef#2{#3/#1}}{}%
  \fi
}
\newcommand{\ResolveCSV}[2]{%
  \def#2{}%
  \ResolveCSVPath{#1}{#2}{.}%
  \ResolveCSVPath{#1}{#2}{data}%
  \ResolveCSVPath{#1}{#2}{csv}%
  \ResolveCSVPath{#1}{#2}{plots}%
  \ResolveCSVPath{#1}{#2}{plot}%
  \ResolveCSVPath{#1}{#2}{fig_data}%
  \ResolveCSVPath{#1}{#2}{fig-data}%
  \ResolveCSVPath{#1}{#2}{tables}%
  \ResolveCSVPath{#1}{#2}{stats}%
  \ResolveCSVPath{#1}{#2}{analysis}%
  \ResolveCSVPath{#1}{#2}{analysis_phase1}%
  \ResolveCSVPath{#1}{#2}{analysis_phase2}%
  \ResolveCSVPath{#1}{#2}{split}%
  \ResolveCSVPath{#1}{#2}{splits}%
  \ResolveCSVPath{#1}{#2}{t1a_splits}%
  \ResolveCSVPath{#1}{#2}{t1b_splits}%
  \ResolveCSVPath{#1}{#2}{t1c_splits}%
  \ResolveCSVPath{#1}{#2}{t1d_splits}%
  \ifx#2\@empty
    \PackageError{t1e-sct}{Could not locate `#1'}%
      {Put the file in the project root or add its folder to \string\ResolveCSV.}%
  \fi
}
\makeatother

\newcommand{\FmtTwo}[1]{\pgfmathprintnumber[fixed,precision=2]{#1}}
\newcommand{\FmtThree}[1]{\pgfmathprintnumber[fixed,precision=3]{#1}}

\begin{document}

\ResolveCSV{plot_src_fig_c_scatter.csv}{\FigCRawPath}
\ResolveCSV{plot_src_fig_c_scatter_stats.csv}{\FigCStatsPath}
\pgfplotstableread[col sep=comma]{\FigCRawPath}\LoadedTable
\pgfplotstableread[col sep=comma]{\FigCStatsPath}\FigCStats
\pgfplotstablegetelem{0}{pooled_spearman_rho}\of{\FigCStats}\edef\PooledRho{\pgfplotsretval}
\pgfplotstablegetelem{0}{pooled_spearman_p}\of{\FigCStats}\edef\PooledP{\pgfplotsretval}
\pgfplotstablegetelem{0}{ols_slope}\of{\FigCStats}\edef\OLSSlope{\pgfplotsretval}
\pgfplotstablegetelem{0}{ols_intercept}\of{\FigCStats}\edef\OLSIntercept{\pgfplotsretval}
\pgfplotstablegetelem{0}{ols_r_squared}\of{\FigCStats}\edef\OLSRsq{\pgfplotsretval}
\gdef\HardScatterData{}
\gdef\MediumScatterData{}
\pgfplotstableforeachcolumnelement{accuracy}\of\LoadedTable\as\ThisAcc{%
  \pgfplotstablegetelem{\pgfplotstablerow}{difficulty}\of{\LoadedTable}%
  \edef\ThisDiff{\pgfplotsretval}%
  \pgfplotstablegetelem{\pgfplotstablerow}{k_star_ratio}\of{\LoadedTable}%
  \edef\ThisRatio{\pgfplotsretval}%
  \ifnum\pdfstrcmp{\ThisDiff}{hard}=0\relax
    \xdef\HardScatterData{\HardScatterData (\ThisAcc,\ThisRatio)}%
  \fi
  \ifnum\pdfstrcmp{\ThisDiff}{medium}=0\relax
    \xdef\MediumScatterData{\MediumScatterData (\ThisAcc,\ThisRatio)}%
  \fi
}
