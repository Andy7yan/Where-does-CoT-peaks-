\newcommand{\FmtTwo}[1]{\pgfmathprintnumber[fixed,precision=2]{#1}}

\begin{document}

\ResolveCSV{plot_src_fig_b_delta.csv}{\FigBRawPath}
\ResolveCSV{plot_src_fig_b_delta_stats.csv}{\FigBStatsPath}
\pgfplotstableread[col sep=comma]{\FigBRawPath}\LoadedTable
\pgfplotstableread[col sep=comma]{\FigBStatsPath}\FigBStats

\gdef\HardDeltaData{}
\gdef\MediumDeltaData{}
\pgfplotstableforeachcolumnelement{delta}\of\LoadedTable\as\ThisDelta{%
  \pgfplotstablegetelem{\pgfplotstablerow}{difficulty}\of{\LoadedTable}%
  \edef\ThisDiff{\pgfplotsretval}%
  \ifnum\pdfstrcmp{\ThisDiff}{hard}=0\relax
    \xdef\HardDeltaData{\HardDeltaData (\ThisDelta,0)}%
  \fi
  \ifnum\pdfstrcmp{\ThisDiff}{medium}=0\relax
    \xdef\MediumDeltaData{\MediumDeltaData (\ThisDelta,0)}%
  \fi
}
