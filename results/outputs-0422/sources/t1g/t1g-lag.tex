% hard: overplot with manual filter by row predicate is cumbersome in raw TeX;

\addplot[
    only marks,
    mark=triangle*,
    mark size=2.6pt,
    draw=HardCol!90!black,
    fill=HardCol!90!black,
    opacity=0.60,
    visualization depends on={\thisrow{difficulty} \as \difficulty},
    filter discard warning=false,
    x filter/.code={
        \edef\tempdiff{\difficulty}
        \def\hardword{hard}
        \ifx\tempdiff\hardword
        \else
            \def\pgfmathresult{inf}
        \fi
    },
] table[
    x expr=\thisrow{L}+0.08,
    y expr=\thisrow{L}-\thisrow{k_star},
] {\LagCSV};

% redraw medium properly with filter
\addplot[
    only marks,
